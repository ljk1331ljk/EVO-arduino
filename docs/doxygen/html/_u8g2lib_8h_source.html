377 /* virtual function for print base class */
378 size_t write(uint8_t v)
379 {
380 uint16_t e = cpp_next_cb(&(u8g2.u8x8), v);
381
382 if (e < 0x0fffe)
383 {
384 u8g2_uint_t delta = u8g2_DrawGlyph(&u8g2, tx, ty, e);
385
386#ifdef U8G2_WITH_FONT_ROTATION
387 switch (u8g2.font_decode.dir)
388 {
389 case 0:
390 tx += delta;
391 break;
392 case 1:
393 ty += delta;
394 break;
395 case 2:
396 tx -= delta;
397 break;
398 case 3:
399 ty -= delta;
400 break;
401 }
402
403 // requires 10 bytes more on avr
404 // tx = u8g2_add_vector_x(tx, delta, 0, u8g2.font_decode.dir);
405 // ty = u8g2_add_vector_y(ty, delta, 0, u8g2.font_decode.dir);
406
407#else
408 tx += delta;
409#endif
410 }
411 return 1;
412 }
413
414 size_t write(const uint8_t *buffer, size_t size)
415 {
416 size_t cnt = 0;
417 while (size > 0)
418 {
419 cnt += write(*buffer++);
420 size--;
421 }
422 return cnt;
423 }
